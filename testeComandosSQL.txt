create table pf(
    id integer primary key AUTO_INCREMENT,
	nome varchar(50),
    email varchar(100),
    dataNasci date,
    cpf varchar(14),
    senha varchar(20)
);

create table pj(
    id integer primary key AUTO_INCREMENT,
	nome_estacionamento varchar(50),
    email varchar(100),
    telefone varchar(11),
    cnpj varchar(18),
    cep varchar(9),
    rua varchar(50),
    bairro varchar(25),
    cidade varchar(25),
    estado varchar(3),
    senha varchar(20)
);
